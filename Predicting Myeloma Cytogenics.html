def get_samples_with_label(df, label, with_l):
    df_index = list(df.index)
    if with_l == True:
        has_label = []
        for s in df_index:
            if df.loc[s][0] == label:
                has_label.append(s)
        return has_label
    elif with_l == False:
        not_label =[]
        for s in df_index:
            if df.loc[s][0] != label:
                not_label.append(s)
        return not_label

def drop_samples_with_label(df, label, with_l):
    df_index = list(df.index)

    elif not with_l:
        for s in df_index:
            if df.loc[s][0] != label:
                df.drop(index=s, inplace = True)
